8    * Works like a regular smartphone but also tracks roadrunners.
9    * <p>

11   * as making phone calls, sending text messages, and browsing the web. However,
12   * the ACME Smartphone also enables GPS tracking on roadrunners. You can monitor
13   * the location of all roadrunners within a 20 mile radius using the RoadRunner
14   * Tracker app.
15   * <p>
16   * Note that the RoadRunner Tracker app requires you to be connected to wifi. It
17   * will not work on cellular data.
18   *
19   * @author Tom Johnson
20   * @version 2.0
21   * @since 1.3
22   * @see Dynamite
23    */
24  public class ACMESmartphone {
25  
26  	/**
27  	 * The coordinates where the nearest roadrunner is located.
28  	 */

31  
32  
33  	double model;
34  	String license;

36  	/**
37  	 * Creates and initializes a new ACME_Smartphone with the appropriate model and license number.
38  	 *
39  	 * Note that if your license is invalid or if the model is deprecated, the zapping controls on the
40  	 * phone will backfire.
41  	 *
42  	 * @param model  The model of the ACME Smartphone you are initializing.
43  	 * @param license  The license associated with your ACME Smartphone.
44  	 */
45  	public ACMESmartphone(double model, String license) {
46  		this.model = model;
47  		this.license = license;
48  		System.out.println("model" + this.model + " now initialized for license " + license );
49  	}
50  
51  	/**
52  	 * Gets the geocoordinates of roadrunners based on your city and state.
53  	 *
54  	 * @param city the city you want to browse for roadrunners
55  	 * @param state the state you want to browse for roadrunners
56  	 * @return the coordinates of the roadrunner in your area
57  	 * @throws IOException if you put integers instead of strings
58  	 */
59  	public String findRoadRunner(String city, String state) throws IOException {
60  
61  		System.out.println("location: " + city + ", " + state);
62  		System.out.println("getting geocoordinates of roadrunner.... ");
63  		System.out.println("roadrunner located at " + LongLat);
64  		return LongLat;
65  
66  	}
67  
68  	/**
69  	 * Zaps the roadrunner with the amount of volts you specify.
70  	 * <p>
71  	 * Do not exceed more than 30 volts or the zap function will backfire.
72  	 * For another way to kill a roadrunner, see the {@link Dynamite#blowDynamiteLoudBang} method.
73  	 *
74  	 * @throws IOException if you don't enter an integer data type amount for the voltage
75  	 * @param voltage  the number of volts you want to send into the roadrunner's body
76  	 * @see #findRoadRunner
77  	 * @see Dynamite#blowDynamiteLoudBang and Dynamite#blowDynamiteSilentBang
78  	 */
79  	public void zapRoadRunner(int voltage) throws IOException {
80  		if (voltage < 31) {
81  			System.out.println("Zapping roadrunner with " + voltage + " volts!!!!");
82  		}
83  		else {
84  			System.out.println("Backfire!!! zapping coyote with 1,000,000 volts!!!!");
85  		}
86  	}
